from django_redis import get_redis_connection

# 从连接池获取一个连接
redis_conn = get_redis_connection('default')

# 设置key
redis_conn.set()
# 获取key
redis_conn.get()

# 清除所有库的所有key
redis_conn.flushall()

# 清除当前库的所有key
redis_conn.flushdb()
