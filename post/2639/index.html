"""""""""""""""""""""""""""""""""""""""""
" return vim mode
"""""""""""""""""""""""""""""""""""""""""
function! Disp_Mode()
    let mode=mode()
    if mode ==# 'v'
        return 'Visual'
    elseif mode ==# 'V'
        return 'V-Line'
    elseif mode ==# ''
        return 'V-Block'
    elseif mode ==# 's'
        return 'Select'
        elseif mode ==# 'S'
        return 'S-Line'
        elseif mode ==# ''
        return 'S-Block'
        elseif mode =~# '\vi'
        return 'Insert'
        elseif mode =~# '\v(R|Rv)'
        return 'Replace'
        else
        return 'Normal'
        endif
endfunction
