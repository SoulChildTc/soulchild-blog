cd /server/packages/eureka-server
cat > src/main/resources/application.yml <<EOF
server:
  port: 8761
eureka:
  # 服务端配置
  server:
    # enable-self-preservation: false  # 关闭自我保护机制
  # 客户端配置
  client:
    # 是否将自己注册到注册中心
    register-with-eureka: false
    # 是否从注册中心获取信息
    fetch-registry: false
  # 控制eureka页面显示内容
  environment: "dev"
  datacenter: "bj-yz"
EOF
